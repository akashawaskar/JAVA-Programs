package AWT_Choice;


import java.applet.Applet;

import java.awt.Choice;

public class Remove_All_Items_From_Choice extends Applet{

Choice language = null;

public void init(){

language = new Choice();

language.add("Java");

language.add("C++");

language.add("VB");

language.add("Perl");

add(language);

language.removeAll();

}

}
