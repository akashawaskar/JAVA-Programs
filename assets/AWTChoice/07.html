package AWT_Choice;


import java.applet.Applet;

import java.awt.Choice;

import java.awt.Graphics;


public class Get_Particular_Item_Of_Choice extends Applet{

Choice language = null;

public void init(){

language = new Choice();

language.add("Java");

language.add("C++");

language.add("VB");

language.add("Perl");

add(language);

}

public void paint(Graphics g){

g.drawString(language.getItem(2) + " is at index 2", 10, 70);

}

}
