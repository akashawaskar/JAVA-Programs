package AWT_Choice;


import java.applet.Applet;

import java.awt.Choice;


public class Insert_Item_At_Specified_Index_In_Choice extends Applet{

public void init(){

language = new Choice();

language.add("Java");

language.add("C++");

language.add("VB");

language.add("Perl");

add(language);

language.insert("PHP",2);

}

}
