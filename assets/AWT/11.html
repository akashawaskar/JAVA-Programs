package AWT;


import java.awt.*;


public class FileDialog_Example {

public static void main (String args[]){

Frame f = new Frame();

f.setVisible(true);

f.setSize(100,100);

FileDialog fd1 = new FileDialog(f, "Example 1",FileDialog.LOAD);

fd1.setVisible(true);

fd1.setSize(100, 100);

fd1.setLayout(new FlowLayout());

Button b1 = new Button("Open");

fd1.add(b1);

}

}
