package AWT;


import java.awt.*;


public class Dialog_Example {

public static void main(String args[]){

Frame f = new Frame("Frame");

f.setVisible(true);

f.setSize(200,200);

FlowLayout fl = new FlowLayout();

f.setLayout(fl);

Label l1 = new Label("Are You Student ??");

Button b1 = new Button("Yes");

Button b2 = new Button("No");

Dialog d = new Dialog(f,"DiaogBox",false);

d.setLayout(new FlowLayout());

d.setVisible(true);

d.setSize(100,100);

d.add(l1);

d.add(b1);

d.add(b2);

}

}
