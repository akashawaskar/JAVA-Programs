package AWT;


import java.awt.BorderLayout;

import java.awt.Button;

import java.awt.Frame;


public class Border_Layout {

public static void main(String args[]){

Frame f = new Frame();

f.setVisible(true);

f.setSize(300, 300);

Button b1 = new Button("Center");

Button b2 = new Button("East");

Button b3 = new Button("West");

Button b4 = new Button("North");

Button b5 = new Button("South");

BorderLayout bl = new BorderLayout();

f.setLayout(bl);

f.add(b1,BorderLayout.CENTER);

f.add(b2,BorderLayout.EAST);

f.add(b3,BorderLayout.WEST);

f.add(b4,BorderLayout.NORTH);

f.add(b5,BorderLayout.SOUTH);

}

}
