package AWT;


import java.applet.*;

import java.awt.*;


public class Card_Layout extends Applet{

Button b1,b2,b3,b4,b5,b6;

Panel main,p1,p2;

int x=0;

GridLayout gl;

BorderLayout bl;

CardLayout cl;

public void init(){

b1=new Button("Window");

b2=new Button("Linux");

add(b1);

add(b2);

main = new Panel();

cl= new CardLayout(5,5);

main.setLayout(cl);

p1=new Panel();

bl=new BorderLayout();

p1.setLayout(bl);

b3=new Button("XP");

p1.add(b3, BorderLayout.CENTER);

b4=new Button ("Vista");

p1.add(b4,BorderLayout.NORTH);

main.add( p1,"Card1");

p2=new Panel();

gl=new GridLayout(2, 1, 5,5);

p2.setLayout(gl);

b5=new Button("Unix");

p2.add( b5);

b6=new Button("Ubntu");

p2.add( b6);

main.add( p2,"Card2");

add(main);

if(x==0){

cl.show(main,"Card1");

}else{

cl.show(main,"Card2");

}

}

}
