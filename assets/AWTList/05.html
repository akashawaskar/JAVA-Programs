package AWT_List;


import java.applet.Applet;

import java.awt.List;


public class Deselect_Item_From_List_Example extends Applet{

public void init(){

List list = new List(5, true);

list.add("One");

list.add("Two");

list.add("Three");

list.add("Four");

list.add("Five");

list.add("Six");

list.add("Seven");

add(list);

list.select(1);

list.deselect(1);

}

}
