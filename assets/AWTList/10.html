package AWT_List;


import java.applet.Applet;

import java.awt.Graphics;

import java.awt.List;


public class Get_Item_From_List_Example extends Applet{

List list = null;

public void init(){

list = new List(5, true);

list.add("One");

list.add("Two");

list.add("Three");

list.add("Four");

list.add("Five");

list.add("Six");

list.add("Seven");

add(list);

}

public void paint(Graphics g){

g.drawString("Item at index 2 is: " +list.getItem(2),10,120);

}

}
