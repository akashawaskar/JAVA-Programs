package AWT_List;


import java.applet.Applet;

import java.awt.Graphics;

import java.awt.List;

import java.awt.event.ItemEvent;

import java.awt.event.ItemListener;


public class Determine_If_Index_Is_Selected_List_Example extends Applet implements ItemListener{

List list = null;

public void init(){

list = new List(5, true);

list.add("One");

list.add("Two");

list.add("Three");

list.add("Four");

list.add("Five");

list.add("Six");

list.add("Seven");

add(list);

list.addItemListener(this);

}

public void paint(Graphics g){

g.drawString("Is index 2 selected? : " + list.isIndexSelected(2), 10, 120);

}

public void itemStateChanged(ItemEvent ie) {

repaint();

}

}
