package AWT_List;


import java.applet.Applet;

import java.awt.Graphics;

import java.awt.List;


public class Set_Multiple_Selection_Mode_For_List_Example extends Applet{

List list = null;

public void init(){

list = new List(5);

list.add("One");

list.add("Two");

list.add("Three");

list.add("Four");

list.add("Five");

list.add("Six");

list.add("Seven");

add(list);

list.setMultipleMode(true);

}

public void paint(Graphics g){

g.drawString("Is Multiple Selection Allowed? " + list.isMultipleMode(), 10, 120);

}

}
