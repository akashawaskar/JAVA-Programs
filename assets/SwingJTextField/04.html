package Swing_JTextField;


import java.awt.FlowLayout;

import javax.swing.JApplet;

import javax.swing.JTextField;


public class Disable_Enable_JTextField_Example extends JApplet{

public void init(){

this.getContentPane().setLayout(new FlowLayout());

eld field = new JTextField("JTextField Disable Enable Example",20);

field.setEnabled(false);

add(field);

}

}
