package SWING;


import java.awt.Container;

import javax.swing.JButton;

import javax.swing.JFrame;

import javax.swing.JTabbedPane;


public class JTabbedPane_Example extends JFrame{

public static void main(String[] args){

JFrame f = new JFrame();

f.setVisible(true);

f.setSize(300, 300);

Container con = f.getContentPane();

JTabbedPane jtp = new JTabbedPane();

JButton jb1 = new JButton("One");

JButton jb2 = new JButton("Two");

JButton jb3 = new JButton("Three");

jtp.addTab("First Tab", jb1);

jtp.addTab("Second Tab", jb2);

jtp.addTab("Second Tab", jb3);

con.add(jtp);

}

}
