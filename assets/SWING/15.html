package SWING;


import java.awt.Container;

import java.awt.FlowLayout;

import javax.swing.JFrame;

import javax.swing.JScrollPane;

import javax.swing.JTree;

import javax.swing.tree.DefaultMutableTreeNode;


public class JTree_Example extends JFrame{

public static void main(String args[]){

JFrame f = new JFrame();

f.setVisible(true);

f.setSize(300, 300);

FlowLayout fl = new FlowLayout();

f.setLayout(fl);

Container con = f.getContentPane();

DefaultMutableTreeNode main = new DefaultMutableTreeNode("Main");

DefaultMutableTreeNode Number = new DefaultMutableTreeNode("Number");

main.add(Number);

DefaultMutableTreeNode one = new DefaultMutableTreeNode("1");

Number.add(one);

DefaultMutableTreeNode two = new DefaultMutableTreeNode("2");

Number.add(two);

DefaultMutableTreeNode three = new DefaultMutableTreeNode("3");

Number.add(three);

DefaultMutableTreeNode alphabet= new DefaultMutableTreeNode("AlphaBet");

main.add(alphabet);

DefaultMutableTreeNode a = new DefaultMutableTreeNode("A");

alphabet.add(a);

DefaultMutableTreeNode b = new DefaultMutableTreeNode("B");

alphabet.add(b);

DefaultMutableTreeNode c = new DefaultMutableTreeNode("C");

alphabet.add(c);

JTree jt = new JTree(main);

JScrollPane jsp = new JScrollPane(jt);

con.add(jsp);

}

}
