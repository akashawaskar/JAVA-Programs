package SWING;


import java.awt.BorderLayout;

import java.awt.CardLayout;

import java.awt.Container;

import java.awt.Dimension;

import javax.swing.JButton;

import javax.swing.JFrame;

import javax.swing.JPanel;


public class Swing_CardLayout {

public static void main(String[] args) {

JFrame f= new JFrame("CardLayout");

f.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

Container con = f.getContentPane();

JPanel jp = new JPanel();

JButton nextButton = new JButton("Next");

jp.add(nextButton);

con.add(jp, BorderLayout.SOUTH);

final JPanel cardPanel = new JPanel();

final CardLayout cl = new CardLayout();

cardPanel.setLayout(cl);

for (int i = 1; i <= 5; i++) {

JButton card = new JButton("Card " + i);

card.setPreferredSize(new Dimension(200, 200));

String cardName = "card" + 1;

cardPanel.add(card, cardName);

}

con.add(cardPanel, BorderLayout.CENTER);

nextButton.addActionListener(e -> cl.next(cardPanel));

f.pack();

f.setVisible(true);

}

}
