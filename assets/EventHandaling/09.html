package Event_Handaling;


import java.awt.Frame;

import java.awt.event.WindowAdapter;

import java.awt.event.WindowEvent;


public class Event_Handaling_With_Frame_Example extends Frame{

Event_Handaling_With_Frame_Example(String title){

super(title);

addWindowListener(new MyWindowAdapter(this));

this.setSize(300,300);

this.setVisible(true);

}

class MyWindowAdapter extends WindowAdapter{

Event_Handaling_With_Frame_Example myWindow = null;

MyWindowAdapter(Event_Handaling_With_Frame_Example myWindow){

this.myWindow = myWindow;

}

public void windowClosing(WindowEvent e) {

myWindow.setVisible(false);

}

}

public static void main(String[] args) {

Event_Handaling_With_Frame_Example myWindow = new Event_Handaling_With_Frame_Example("Window with event Example");

}
