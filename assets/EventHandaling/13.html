package Event_Handaling;


import java.applet.Applet;

import java.awt.Graphics;

import java.awt.event.*;


public class MouseListener_Example extends Applet implements MouseListener{

String msg="";

int x=0,y=0;

public void init(){

addMouseListener(this);

}

public void mouseEntered(MouseEvent e){

msg="Mouse Entered ";

repaint();

x=e.getX();

y=e.getY();

msg=msg+" Position at "+x+","+y;

}

public void mouseClicked(MouseEvent e){

msg="Mouse Clicked ";

repaint();

x=e.getX();

y=e.getY();

msg=msg+" Position at "+x+","+y;

}

public void mousePressed(MouseEvent e){

msg="Mouse Pressed ";

repaint();

x=e.getX();

y=e.getY();

msg=msg+" position at "+x+","+y;

}

public void mouseReleased(MouseEvent e){

msg="Mouse Released ";

repaint();

x=e.getX();

y=e.getY();

msg=msg+" Position at "+x+","+y;

}

public void mouseExited(MouseEvent e){

msg="Mouse Exited ";

repaint();

x=e.getX();

y=e.getY();

msg=msg+" Position at "+x+","+y;

}

public void paint(Graphics g){

g.drawString(msg, x, y);

}

}
