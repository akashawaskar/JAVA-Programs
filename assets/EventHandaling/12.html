package Event_Handaling;


import java.applet.Applet;

import java.awt.*;

import java.awt.event.ActionEvent;

import java.awt.event.ActionListener;


public class Event_Handaling_With_MenuBar_Example extends Applet implements ActionListener{

MenuBar mb;

Menu m1,m2;

MenuItem n,o,s,q;

Frame f;

FileDialog fd;

String msg="";

public void init() {

f=new Frame();

f.setVisible(true);

f.setSize(300, 300);

mb=new MenuBar();

f.setMenuBar(mb);

m1=new Menu("File");

m2=new Menu("Edit");

mb.add(m1);

mb.add(m2);

n=new MenuItem("New");

o=new MenuItem("Open");

s=new MenuItem("Save");

q=new MenuItem("Quit");

m1.add(n);

m1.add(o);

m1.add(s);

m1.add(q);

n.addActionListener( this);

o.addActionListener( this);

s.addActionListener( this);

q.addActionListener( this);

}

public void actionPerformed(ActionEvent e){

if(e.getSource()==n){

msg="You Pressed New";

}else if(e.getSource()==o){

fd=new FileDialog(f,"Dialog",FileDialog.LOAD);

fd.setVisible(true);

}else if(e.getSource()==s){

fd=new FileDialog(f,"Dialog",FileDialog.SAVE);

fd.setVisible(true);

}else{

msg="You Pressed Quit";

}

repaint();

}

public void paint(Graphics g){

g.drawString(msg, 80, 150);

}

}
