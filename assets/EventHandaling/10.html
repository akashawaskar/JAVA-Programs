package Event_Handaling;


import java.applet.Applet;

import java.awt.Button;

import java.awt.GridLayout;

import java.awt.Label;

import java.awt.TextField;

import java.awt.event.ActionEvent;

import java.awt.event.ActionListener;


public class Event_Handaling_With_Label_Example extends Applet implements ActionListener{

Label l1,l2,l3;

Button b1,b2;

TextField t1,t2,t3;

public void init(){

l1=new Label("Name");

l2=new Label("Password");

l3=new Label("Result");

b1=new Button("Login");

b2=new Button("Cancel");

t1=new TextField();

t2=new TextField();

t3=new TextField();

GridLayout gl=new GridLayout(4,2);

setLayout(gl);

b1.addActionListener(this);

b2.addActionListener(this);

add(l1);

add(t1);

add(l2);

add(t2);

add(l3);

add(t3);

add(b1);

add(b2);

}

public void actionPerformed(ActionEvent e){

String n1,n2;

String n=t1.getText();

String p=t2.getText();

if(e.getSource()==b2){

t3.setText("Operation Cancel");

}else if(n.equals("Akash") && p.equals("1234")){

t3.setText("Login Sucessfull");

}else{

t3.setText("Login Fail");

}

}

}
