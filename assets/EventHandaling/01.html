package Event_Handaling;


import java.applet.*;

import java.awt.*;

import java.awt.event.*;


public class Event_Handaling_With_Button_Example1 extends Applet implements ActionListener{

String msg="";

Button b1,b2;

public void init(){

b1=new Button("One");

b2=new Button("Two");

setLayout(new FlowLayout());

b1.addActionListener(this);

b2.addActionListener(this);

add(b1);

add(b2);

}

public void actionPerformed(ActionEvent e){

msg="You Pressed ";

String s=e.getActionCommand();

if(s.equals("One"))

{

msg=msg+s;

repaint();

}

else

{

msg=msg+s;

repaint();

}

}

public void paint(Graphics g){

g.drawString(msg, 20, 160);

}

}
