package Event_Handaling;


import java.applet.Applet;

import java.awt.*;

import java.awt.event.*;


public class Event_Handaling_With_CheckBox_Example extends Applet implements ActionListener{

Checkbox c1,c2,c3;

Button b1,b2;

String msg="";

int total;

public void init() {

c1=new Checkbox("Breakfatt");

c2=new Checkbox("Launch");

c3=new Checkbox("Dinner");

b1=new Button("Bill");

b2=new Button("Cancel");

add(c1);

add(c2);

add(c3);

add(b1);

add(b2);

b1.addActionListener(this);

b2.addActionListener(this);

}

public void actionPerformed(ActionEvent e){

if(e.getSource()==b1){

if(c1.getState()){

total=total+50;

}

if(c2.getState()){

total=total+150;

}

if(c3.getState()){

total=total+100;

}

msg="Total Bill "+total+"";

}else{

msg="Opration Cancelled";

}

repaint();

}

public void paint(Graphics g){

g.drawString(msg,60, 100);

}

}
