package Event_Handaling;


import java.awt.Button;

import java.awt.GridLayout;

import java.awt.Label;

import java.awt.TextField;

import java.awt.event.*;

import java.applet.*;

import java.awt.*;


public class Event_Handaling_With_Label_Example2 extends Applet implements ActionListener {

Label l1,l2,l3;

Button b1,b2;

TextField t1,t2,t3;

public void init(){

l1=new Label("Num1");

l2=new Label("Num2");

l3=new Label("Result");

b1=new Button("Addition");

b2=new Button("Substraction");

t1=new TextField();

t2=new TextField();

t3=new TextField();

GridLayout gl=new GridLayout(4,2);

setLayout(gl);

b1.addActionListener(this);

b2.addActionListener(this);

add(l1);

add(t1);

add(l2);

add(t2);

add(l3);

add(t3);

add(b1);

add(b2);

}

public void actionPerformed(ActionEvent e){

String n1,n2;

int x,y,z;

n1=t1.getText();

n2=t2.getText();

x=Integer.parseInt(n1);

y=Integer.parseInt(n2);

if(e.getSource()==b1){

z=x+y;

t3.setText(z+"");

}else{

z=x-y;

t3.setText(z+"");

}

}

}
