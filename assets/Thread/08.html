package Thread;


public class Thread_Synchronization_A extends Thread{

public void run(){

for(int i=0;i<5;i++){

System.out.println("Thread A="+i);

}

}

}


package Thread;


public class Thread_Synchronization_B extends Thread{

public int total;

public void run(){

synchronized(this){

for(int i=0;i<100;i++){

total+=i;

}

notify();

}

}

}



package Thread;

public class Thread_Synchronization_C extends Thread{

public void run(){

for(int j=0;j<5;j++){

System.out.println("Thread c="+j);

}

}

}


package Thread;


import Thread.Thread_Synchronization_A;

import Thread.Thread_Synchronization_B;

import Thread.Thread_Synchronization_C;


public class Thread_SynchronizationTest {

public static void main(String[] args) {

Thread_Synchronization_A A =new Thread_Synchronization_A();

Thread_Synchronization_B B =new Thread_Synchronization_B();

Thread_Synchronization_C C =new Thread_Synchronization_C();

B.start();

A.start();

C.start();

synchronized(B){

try{

System.out.println("Waiting for b to complete");

B.wait();

}catch(InterruptedException e){

e.printStackTrace();

}

System.out.println("Total is"+B.total);

}

}

}
