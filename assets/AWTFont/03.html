package AWT_Font;


import java.applet.Applet;

import java.awt.Font;

import java.awt.Graphics;


public class Create_Italic_Font_Example extends Applet{

public void paint(Graphics g) {

Font myFont = new Font("Courier", Font.ITALIC,20);

g.setFont(myFont);

g.drawString("Italic Font Example By Akash Awaskar", 10, 50);

}

}
