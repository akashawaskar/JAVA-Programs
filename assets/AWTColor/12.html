package AWT_Color;


import java.applet.Applet;

import java.awt.Color;

import java.awt.Graphics;

public class Get_Transparency_Mode_Of_Color_Example extends Applet{

public void paint(Graphics g){

Color customColor = new Color(255,10,10);

int transparency = customColor.getTransparency();

g.drawString("Transparency mode : " + transparency, 10,50);

}

}
