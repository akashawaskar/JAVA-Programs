package AWT_Color;


import java.applet.Applet;

import java.awt.Color;

import java.awt.Graphics;


public class Convert_RGB_To_HSB_Color_Example extends Applet{

public void paint(Graphics g){

float[] hsbValues = new float[3];

hsbValues = Color.RGBtoHSB(255,10,10,hsbValues);

float hue, saturation, brightness;

hue = hsbValues[0];

saturation = hsbValues[1];

brightness = hsbValues[2];

g.drawString("Hue: " + hue + ", Saturation:" + saturation + ", Brightness:" + brightness,10,50);

}

}
